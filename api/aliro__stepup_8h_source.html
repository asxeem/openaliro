1// Aliro step-up (Access Document) phase: builds the mdoc DeviceRequest, unwraps and decrypts the
2// SessionData DeviceResponse, decodes the CBOR document per spec 7.2/8.4.2, and runs the six-step
3// Access Document verification of spec 7.4. Reference-completeness codec + verifier; the verdict is
4// logged and stored, never gates the unlock (the provisioned trust store remains the sole gate).
5/*
6 * Copyright (c) 2026 asxeem
7 * SPDX-License-Identifier: ISC
8 *
9 * aliro_stepup — the Aliro §8.4 step-up phase: the ISO 18013-5 (mdoc) document
10 * exchange the Reader MAY run in the standard phase to obtain an Access or
11 * Revocation Document. Three concerns, split across two translation units so the
12 * wire-facing decoder fuzzes with no crypto dependency:
13 *
14 * aliro_stepup_parse.c pure CBOR decode + DeviceResponse structural parse
15 * (Table 7-1/7-2/8-22). No crypto, no allocation; every
16 * field is a bounds-checked slice of the caller's buffer.
17 * aliro_stepup.c the DeviceRequest builder, the ENVELOPE / GET RESPONSE
18 * APDU codec, the SessionData seal/open (reusing the
19 * aliro_secchan AES-256-GCM channel under StepUpSK), and
20 * the §7.4 verifier (digest recompute + validity + the
21 * issuer-signature check via an injected ES256 verify).
22 *
23 * The ES256 primitive is passed in (aliro_stepup_verify_ctx.ecdsa_verify) so this
24 * module carries no elliptic-curve dependency: the target wires the PSA-backed
25 * aliro_ecdsa_p256_verify, the host KAT injects its own.
26 *
27 * Provenance: clean-room. Structures from the Aliro v1.0 spec (§7, §8.4, §14.6)
28 * and ISO 18013-5; the code is original.
29 */

58/* ---- requester (§8.4.2) ----
59 * Build the fixed Access-Document DeviceRequest CBOR (Table 8-21): docType
60 * "aliro-a", namespace "aliro-a", requesting the given element identifiers
61 * (intent-to-retain true). With elems=NULL/n_elems=0, requests element2+element4
62 * (the §14.6 example request). Returns 0 and sets *out_len, or -1 on overflow. */

64 size_t cap, size_t *out_len);

66/* Seal a DeviceRequest into a SessionData message {"data": bstr(ct||tag)} and

71/* Open a SessionData message: unwrap {"data": bstr}, AES-256-GCM-open under the

77/* ---- ENVELOPE / GET RESPONSE APDUs (§8.4.4) ----

105struct aliro_stepup_doc {
106 int have_document; /* 0 = DeviceResponse carried no documents (device declined) */
107 int status; /* DeviceResponse "3" */
108
109 char doc_type[ALIRO_STEPUP_ID_MAX]; /* Documents[].docType ("5") */
110 char name_space[ALIRO_STEPUP_ID_MAX]; /* the single issuerSigned namespace */
111
112 /* IssuerAuth COSE_Sign1 [protected, unprotected, payload, signature]. */
113 const uint8_t *protected_hdr; /* content of the protected bstr */
114 size_t protected_len;
115 const uint8_t *kid; /* unprotected label 4, or NULL */
116 size_t kid_len;
117 const uint8_t *x5chain; /* unprotected label 33, or NULL (raw item bytes) */
118 size_t x5chain_len;
119 const uint8_t *payload; /* content of the payload bstr = 24(bstr(MSO)) */
120 size_t payload_len;
121 const uint8_t *signature; /* 64-byte r||s */
122
123 /* MobileSecurityObject (Table 7-1). */
124 char digest_alg[ALIRO_STEPUP_ID_MAX]; /* "SHA-256" */
125 char mso_doc_type[ALIRO_STEPUP_ID_MAX];
126 struct aliro_stepup_digest digests[ALIRO_STEPUP_MAX_DIGESTS];
127 size_t n_digests;
128
129 /* validityInfo tdates as parsed epoch seconds (UTC). */
130 int have_signed, have_valid_from, have_valid_until;
131 int64_t signed_epoch, valid_from_epoch, valid_until_epoch;
132 int have_iteration;
133 uint64_t iteration;
134 int time_verification_required;
135
136 /* disclosed IssuerSignedItems. */
137 struct aliro_stepup_item items[ALIRO_STEPUP_MAX_ITEMS];
138 size_t n_items;
139};

147struct aliro_stepup_issuer {

153struct aliro_stepup_verify_ctx {
154 const struct aliro_stepup_issuer *issuers;
155 size_t n_issuers;
156 int time_valid; /* reader holds a trusted clock (timesync) */
157 int64_t now_epoch; /* current UTC seconds; used only when time_valid */
158 uint64_t access_iteration; /* stored AccessIteration for this issuer (default 0) */
159 const char *expected_doctype;
160 /* ES256 over msg (hashing internal), pub = 65-byte point, sig = 64-byte r||s.
161 * Returns 0 on a valid signature. Target: aliro_ecdsa_p256_verify. */
162 int (*ecdsa_verify)(const uint8_t pub[65], const uint8_t *msg, size_t msg_len,
163 const uint8_t sig[64]);
164};

166struct aliro_stepup_verdict {
167 int valid; /* all applicable steps passed AND >=1 valid element */
168 int reject_step; /* 0 = accepted; else the first §7.4 step that failed (1..6) */
169
170 int issuer_key_found; /* step 1 */
171 int issuer_chain_validated; /* 0 for x5chain: EE key used, chain NOT validated (ref limit)
172 */
173 int sig_ok; /* step 2 */
174 int digests_ok; /* step 3: every disclosed item matched its valueDigest */
175 int doctype_ok; /* step 4 */
176 int time_ok; /* step 5 */
177 int iteration_ok; /* step 6 */
178 size_t valid_elements; /* disclosed items whose digest verified */
179};

181/* Run §7.4 over a parsed document. Never gates access; fills *verdict for the
182 * caller to log/store. Returns 0 if verdict->valid, <0 otherwise (same info as
183 * verdict->reject_step; the return is a convenience, not an access decision). */
184int aliro_stepup_verify(const struct aliro_stepup_doc *doc,
185 const struct aliro_stepup_verify_ctx *ctx,
186 struct aliro_stepup_verdict *verdict);

188/* Convenience: open the SessionData response, parse, and verify in one call.
189 * Fills *verdict; returns 0 on verdict->valid, <0 on any decrypt/parse/verify
190 * failure (the worker logs the verdict regardless of the return). scratch must
191 * hold the decrypted DeviceResponse (>= sd_len). */
192int aliro_stepup_run(struct aliro_secchan *sc, const uint8_t *sd_resp, size_t sd_len,
193 const struct aliro_stepup_verify_ctx *ctx, uint8_t *scratch,
194 size_t scratch_cap, struct aliro_stepup_doc *doc,
195 struct aliro_stepup_verdict *verdict);

197/* ---- ESP worker seam (implemented per-platform; see aliro_stepup_worker.c) ----
198 * Copies the collected SessionData response + keys + verify inputs and runs
199 * aliro_stepup_run() off the BLE-host task, so parse/verify never touches the
200 * auth segment or the ranging arm window. Returns 0 if queued. */
201struct aliro_stepup_job {
202 uint8_t sk_reader[ALIRO_SESSION_KEY_LEN];
203 uint8_t sk_device[ALIRO_SESSION_KEY_LEN];
204 uint8_t issuer_pub[65];
205 uint8_t issuer_kid[16];
206 size_t issuer_kid_len;
207 int have_issuer;
208 int time_valid;
209 int64_t now_epoch;
210 uint16_t conn_handle;
211 size_t sd_len;
212 uint8_t sd[2048]; /* SessionData response (x5chain-cert headroom) */
213};
214int aliro_stepup_worker_submit(const struct aliro_stepup_job *job);
