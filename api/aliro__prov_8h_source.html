1// Persistent reader provisioning storage: identity and credential trust anchors saved to and
2// loaded from NVS.
3// Declares aliro_prov_store for committing an identity/trust pair to NVS, and struct
4// aliro_trust_store, the set of trusted credential public keys against which a presented
5// credential is authenticated.
6/*
7 * Copyright (c) 2026 asxeem
8 * SPDX-License-Identifier: ISC
9 *
10 * aliro_prov — the reader provisioning seam (Phase 3.4): the reader's own
11 * identity (a stable reader identifier + P-256 signing key) and the trust store
12 * of credential public keys it authenticates presented credentials against.
13 * NVS-backed on target, with a clearly-marked dev fallback so the credential-auth
14 * transaction can be driven at bench before Phase-4 Matter provisioning writes a
15 * real identity.
16 *
17 * Split like aliro_crypto: the (de)serialisation + dev default + trust logic is
18 * portable and host-KAT'd (aliro_prov.c); the NVS load/store is target-only
19 * (aliro_prov_nvs.c, not compiled on host). Provenance: clean-room; original.
20 */

31#define ALIRO_READER_ID_LEN 32u
32#define ALIRO_READER_PRIV_LEN 32u
33#define ALIRO_CRED_PUB_LEN 65u /* uncompressed P-256 point: 0x04 | X | Y */
34#define ALIRO_TRUST_MAX 4u /* trusted credential keys the store holds */
35#define ALIRO_GRK_LEN 16u /* group resolving key (Aliro BLE-UWB adv tag) */
36#define ALIRO_KPERSISTENT_LEN 32u /* per-credential expedited-fast key (§8.3.1.13) */
37
38/*
39 * The reader's provisioned identity. reader_id rides AUTH0 and both ECDSA
40 * transcripts (tag 0x4D); sign_priv signs the reader-usage transcript. is_dev
41 * marks the built-in bench identity, never a real deployment.
42 */
43struct aliro_reader_identity {
44 uint8_t reader_id[ALIRO_READER_ID_LEN];
45 uint8_t sign_priv[ALIRO_READER_PRIV_LEN];
46 uint8_t grk[ALIRO_GRK_LEN]; /* group resolving key; all-zero if none */
47 bool is_dev;
48};
49
50/*
51 * Trusted credential public keys. A presented credential authenticates only if
52 * its key is in here (or the store is empty and dev policy allows it). A raw-key
53 * allowlist is the interim seam; real issuer-chain validation is the Phase-4
54 * refinement that plugs in at aliro_prov_trust_check.
55 */
56struct aliro_trust_store {
57 uint8_t count;
58 uint8_t cred_pub[ALIRO_TRUST_MAX][ALIRO_CRED_PUB_LEN];
59 /* Expedited-fast state: bit i of kp_valid set = kpersistent[i] holds the
60 * Kpersistent agreed with cred_pub[i] in its last standard phase. */
61 uint8_t kp_valid;
62 uint8_t kpersistent[ALIRO_TRUST_MAX][ALIRO_KPERSISTENT_LEN];
63};
64
65/* Serialised blob v3: magic(4) ver(1) flags(1) reader_id(32) sign_priv(32)
66 * grk(16) count(1), count * cred_pub(65), kp_valid(1), count * kpersistent(32).
67 * (v2 ended at the cred_pub array; v1 also had no grk. Both still parsed.) */
68#define ALIRO_PROV_BLOB_HDR 6u
69#define ALIRO_PROV_BLOB_MAX \
70 (ALIRO_PROV_BLOB_HDR + ALIRO_READER_ID_LEN + ALIRO_READER_PRIV_LEN + ALIRO_GRK_LEN + 1u + \
71 (size_t)ALIRO_TRUST_MAX * ALIRO_CRED_PUB_LEN + 1u + \
72 (size_t)ALIRO_TRUST_MAX * ALIRO_KPERSISTENT_LEN)
73
74/* ---- portable core (aliro_prov.c) --------------------------------------- */
75
76/* Populate the built-in clearly-marked dev identity + an empty trust store. */
77void aliro_prov_dev_default(struct aliro_reader_identity *id, struct aliro_trust_store *ts);
78
79/* Serialise identity+trust to a self-describing blob. 0 + *out_len on success,
80 * -1 on overflow (cap < the assembled length). */
81int aliro_prov_serialize(const struct aliro_reader_identity *id, const struct aliro_trust_store *ts,
82 uint8_t *out, size_t cap, size_t *out_len);
83
84/* Parse a blob written by aliro_prov_serialize. 0 on success; -1 if malformed
85 * (bad magic/version/length/count). Outputs are untouched on failure. */
86int aliro_prov_deserialize(const uint8_t *buf, size_t len, struct aliro_reader_identity *id,
87 struct aliro_trust_store *ts);
88
89/* Trust decision for a presented credential public key:
90 * 0 trusted (cred_pub matches a stored key)
91 * 1 no-anchors (store empty; caller applies dev-open policy)
92 * -1 rejected (store non-empty and no match). */
93int aliro_prov_trust_check(const struct aliro_trust_store *ts,
94 const uint8_t cred_pub[ALIRO_CRED_PUB_LEN]);
95
96/* Add a credential key to the store. 0 added; 1 already present (dedup); -1 full
97 * or the point is not an uncompressed P-256 point (leading byte != 0x04). */
98int aliro_prov_trust_add(struct aliro_trust_store *ts, const uint8_t cred_pub[ALIRO_CRED_PUB_LEN]);
99
100/* Index of a credential key in the store, or -1 if not present. */
101int aliro_prov_trust_find(const struct aliro_trust_store *ts,
102 const uint8_t cred_pub[ALIRO_CRED_PUB_LEN]);
103
104/* Bind a Kpersistent (§8.3.1.13) to the credential at idx (from
105 * aliro_prov_trust_find), replacing any earlier one. 0 on success; -1 if idx is
106 * not a stored credential. */
107int aliro_prov_kpersistent_set(struct aliro_trust_store *ts, int idx,
108 const uint8_t kp[ALIRO_KPERSISTENT_LEN]);
109
110/* ---- target NVS backend (aliro_prov_nvs.c) ------------------------------ */
111
112/* Load identity+trust from NVS; on absence or a malformed blob fall back to the
113 * dev default (leaving NVS untouched). Always yields a usable identity.
114 * 0 a stored blob was loaded
115 * 1 the dev default was used (nothing stored)
116 * -1 an NVS error occurred; the dev default was used. */
117int aliro_prov_load(struct aliro_reader_identity *id, struct aliro_trust_store *ts);
118
119/* Persist identity+trust to NVS. 0 on success, negative on an NVS error. */
120int aliro_prov_store(const struct aliro_reader_identity *id, const struct aliro_trust_store *ts);
