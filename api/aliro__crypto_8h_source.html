7 * aliro_crypto — the credential-auth cryptography for the Aliro reader: the
8 * P-256 / AES-256-GCM / SHA-256 suite and the key-derivation schedule that
9 * turns a passed credential authentication into the 32-byte URSK (the UWB
10 * ranging root) plus the secure-channel keys.
11 *
12 * This is Phase 3.1: the primitives and the key schedule, host-KAT verifiable.
13 * The transaction state machine that drives it (AUTH0/AUTH1, EXCHANGE) is 3.2;
14 * the handoff of the URSK into woz_uwb_start_aliro(cfg) is 3.3.
15 *
16 * Provenance: clean-room. The suite is standard (mbedTLS-PSA + a portable
17 * SHA-256/KDF core). The wire/derivation facts come from the project's
18 * reverse-engineering notes, not from any vendor source.

51/*
52 * ---- Credential-auth key schedule (standard/ECDH path) -------------------
53 *
54 * Stage 1: Z = SHA-256( shared_secret(32) | 0x00000001 | txid(16) ), the
55 * single-block ANSI-X9.63 concat KDF over the raw ECDH shared secret. Z is the
56 * IKM for every stage-2 HKDF.
57 */
58void aliro_crypto_derive_z(const uint8_t shared_secret[ALIRO_SHARED_SECRET_LEN],
59 const uint8_t txid[ALIRO_TXID_LEN], uint8_t z[32]);
60
61/*
62 * Stage 2: block = HKDF-SHA256(salt, IKM=z, info=device_pub_x(32), L=160). The
63 * salt is the CreateSalt transcript (aliro_salt_build). Returns 0 on success.
64 */
65int aliro_crypto_derive_block(const uint8_t z[32], const uint8_t *salt, size_t salt_len,
66 const uint8_t device_pub_x[ALIRO_EC_PUBX_LEN],

69/*
70 * Single 32-byte keyed derivation off z: Kpersistent (salt type 2) and the

77/*
78 * Split the 160-byte block into the two directional session keys + URSK. The
79 * reference derives up to two optional "shared" keys (C/D) alongside; a config
80 * flag shifts which segments are the symmetric keys. with_c=1: enc=S0, dec=S1;
81 * with_c=0: enc=S1, dec=S2 (S0 unused). URSK = S4 (offset 128) either way.

87/*
88 * ---- Secure channel (AES-256-GCM, directional per-message counters) ------
89 *
90 * Nonce = 8-byte big-endian direction (0 outbound/seal, 1 inbound/open) followed
91 * by a 4-byte big-endian per-direction counter. Separate seal/open counters,
92 * start at 0, no wrap. SessionCrypto sends no AAD; the BLE channel authenticates
93 * a 4-byte AAD (caller-supplied here).
94 */
95struct aliro_secchan {
96 uint8_t enc_key[ALIRO_SESSION_KEY_LEN];
97 uint8_t dec_key[ALIRO_SESSION_KEY_LEN];
98 uint32_t enc_ctr;
99 uint32_t dec_ctr;
100};

116/*
117 * ---- Aliro message security (§11.8): ranging/notification SDUs -----------
118 *
119 * Proto-1/2/3 SDUs (UWB Ranging Service M1-M4, Notification, Supplementary) ride
120 * a SEPARATE AES-256-GCM channel from the AP secure channel: BleSKReader/
121 * BleSKDevice keys (HKDF off BleSK = block offset 96), fresh per-direction
122 * counters starting at 1, and the 4-byte header (with the PLAINTEXT payload
123 * length) as AAD. Wire form: [proto][id][len_be16][encrypted_payload||16B tag],
124 * where len_be16 = plaintext length + 16. Reuse struct aliro_secchan for it
125 * (enc=BleSKReader, dec=BleSKDevice; aliro_secchan_init sets both counters to 1).
126 */
127#define ALIRO_BLESK_OFFSET 96u /* BleSK = block[96 .. 127] (§8.3.1.12/.13) */
128
129/* Derive BleSKReader + BleSKDevice from the 160-byte block per §11.8.1:
130 * HKDF-SHA256(ikm=BleSK, info="BleSKReader"/"BleSKDevice", L=32,
131 * salt = reader_supported_versions || user_device_selected_version). 0 on ok. */
132int aliro_crypto_derive_ble_keys(const uint8_t block[ALIRO_KEY_BLOCK_LEN], const uint8_t *salt,
133 size_t salt_len, uint8_t ble_reader[ALIRO_SESSION_KEY_LEN],
134 uint8_t ble_device[ALIRO_SESSION_KEY_LEN]);

136/* Seal an engine plaintext message [proto][id][len_plain_be16][payload] into the
137 * on-wire [proto][id][(len_plain+16)_be16][ct||tag], sealed under sc with the

139int aliro_msg_seal(struct aliro_secchan *sc, const uint8_t *plain, size_t plain_len, uint8_t *wire,

147/*
148 * ---- CreateSalt transcript builder --------------------------------------
149 *
150 * Builds the stage-2 HKDF salt byte-exact to Aliro §8.3.1.13 (salt_volatile for
151 * the SESSION/standard type):
152 * span_s1(reader_group_identifier_key.x, 32) || label(12) || reader_id(32) ||
153 * interface_byte(1) || 0x5C || 0x02 || protocol_version(2) ||
154 * reader_value(reader ephemeral pub X, 32) || txid(16) ||
155 * flag(exp_phase_type||user_auth_policy, 2) || a5_tlv(0xA5 proprietary info).
156 * For the fast/persistent types a trailing s3opt (Access Credential public key X)
157 * follows the a5_tlv. Returns 0 on success and the assembled length in *out_len.
158 */
159enum aliro_salt_type {
160 ALIRO_SALT_CRYPTOGRAM = 0, /* label "VolatileFast" */
161 ALIRO_SALT_SESSION = 1, /* label "Volatile****" */
162 ALIRO_SALT_KPERSISTENT = 2 /* label "Persistent**" */
163};
